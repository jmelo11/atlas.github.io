109 // Initialize distances and predecessors

111 std::unordered_map<size_t, size_t> predecessors;
112 for (const auto& [key, _] : exchangeRateMap_) {
113 distances[std::get<0>(key)] = std::numeric_limits<adouble>::infinity();

117 }

119 distances[startCurrency] = 0.0; // Set the first currency's distance to 0
120
121 // Bellman-Ford algorithm
122 size_t numCurrencies = distances.size();
123 size_t lastUpdatedCurrency = 0;
124 for (size_t i = 0; i < numCurrencies - 1; ++i) {
125 for (const auto& [key, value] : exchangeRateMap_) {

127 auto dest = std::get<1>(key);
128 adouble newDistance = distances[source] - std::log(value);
129 if (newDistance < distances[dest]) {
130 distances[dest] = newDistance;
131 predecessors[dest] = source;
132 lastUpdatedCurrency = dest;
133 }
134 }
135 }
136
137 // Check for negative cycles
138 for (const auto& [key, value] : exchangeRateMap_) {
139 auto source = std::get<0>(key);
140 auto dest = std::get<1>(key);
141 adouble newDistance = distances[source] - std::log(value);
142 if (newDistance < distances[dest]) {
143 // Reconstruct the cycle
144 std::vector<size_t> cycle;
