58 void simulateDiscounts(const Date& evalDate, MarketData<adouble>& md) const {
59 for (size_t i = 0; i < this->marketRequest_.dfs.size(); ++i) {
60 const auto& request = this->marketRequest_.dfs.at(i);
61 size_t idx = request.curve_;
62 const Date& date = request.date_;

65 adouble df;

83 void simulateForwardRates(const Date& evalDate, MarketData<adouble>& md) const {
84 for (size_t i = 0; i < this->marketRequest_.fwds.size(); ++i) {
85 const auto& request = this->marketRequest_.fwds.at(i);

87 const Date& startDate = request.startDate_;
88 const Date& endDate = request.endDate_;
89
90 const YieldTermStructure<adouble>& curve = marketStore_.curveContext(idx).object();
91 const RateIndex<adouble>& index = marketStore_.rateIndexContext(idx).object();
92
93 adouble fwd;
94 if (evalDate <= startDate) {
95 fwd = curve.forwardRate(startDate, endDate, index.dayCounter(), index.rateCompounding(), index.rateFrequency());
96 } else {
97 fwd = index.getFixing(startDate);
98 }
99 md.fwds.push_back(fwd);

115 const Date& date = request.date_;
116
117 adouble fx = marketStore_.exchange(ccy1, ccy2);
118 if (date != Date()) {
119 size_t ccy1RFC = marketStore_.riskFreeCurveIdx(ccy1);
120 size_t ccy2RFC = marketStore_.riskFreeCurveIdx(ccy2);
121 const YieldTermStructure<adouble>& ccy1Curve = marketStore_.curveContext(ccy1RFC).object();
122 const YieldTermStructure<adouble>& ccy2Curve = marketStore_.curveContext(ccy2RFC).object();
123 fx *= ccy2Curve.discount(date) / ccy1Curve.discount(date);
124 }
125 md.fxs.push_back(fx);
