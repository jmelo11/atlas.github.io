104 void indexCashflow(Flow& cashflow) {
105 if (!cashflow.hasDiscountContext()) { throw std::runtime_error("Cashflow does not have a discount curve context."); }
106 size_t curveIdx = cashflow.discountContextIdx();
107 const Date& paymentDate = cashflow.paymentDate();
108 evaluationDates_.insert(paymentDate);
109
110 MarketRequest::DiscountFactor df(curveIdx, paymentDate);
111
112 // If the discount factor has not been indexed, add it to the vector
113 if (dfsMap_.find(df) == dfsMap_.end()) {
114 dfsVector_.push_back(df);
115 dfsMap_[df] = dfsVector_.size() - 1;
116 }
117 cashflow.dfIdx(dfsMap_[df]);
118
119 // fwd rate
120 if constexpr (std::is_same_v<Flow, FloatingRateCoupon<adouble>>) {
